using System;
using System.Collections.Generic;
using System.Linq;
using System.Reflection;
using System.Reflection.Emit;
using HarmonyLib;
using Unity.Collections.LowLevel.Unsafe;

namespace MapExtPDX.MapExt.ReBurstSystem.Core
{
	public static class GenericJobReplacePatch
	{
		private class MethodPatchContext
		{
			public Dictionary<Type, Type> JobReplacements { get; } = new Dictionary<Type, Type>();

			public Dictionary<FieldInfo, FieldInfo> FieldReplacements { get; } = new Dictionary<FieldInfo, FieldInfo>();
		}

		private static readonly string ModName;

		private static readonly string patchTypeName;

		private static Dictionary<MethodBase, MethodPatchContext> activePatchContexts;

		private static readonly object _contextLock;

		public static void Info(string message)
		{
			Mod.Info("[" + GenericJobReplacePatch.ModName + "." + GenericJobReplacePatch.patchTypeName + "] " + message);
		}

		public static void Warn(string message)
		{
			Mod.Warn("[" + GenericJobReplacePatch.ModName + "." + GenericJobReplacePatch.patchTypeName + "] ⚠\ufe0f " + message);
		}

		public static void Error(string message)
		{
			Mod.Error("[" + GenericJobReplacePatch.ModName + "." + GenericJobReplacePatch.patchTypeName + "] ❌ " + message);
		}

		public static void Error(Exception e, string message)
		{
			Mod.Error(e, "[" + GenericJobReplacePatch.ModName + "." + GenericJobReplacePatch.patchTypeName + "] ❌ " + message);
		}

		public static void AddReplacementToContext(MethodBase method, Type originalJobType, Type replacementJobType)
		{
			if (method == null || originalJobType == null || replacementJobType == null)
			{
				return;
			}
			JobFieldValidator.ValidationResult validation = JobFieldValidator.ValidateJobReplacement(originalJobType, replacementJobType);
			if (!validation.IsValid)
			{
				GenericJobReplacePatch.Error("注册失败 | Job 结构不兼容 " + originalJobType.Name + " -> " + replacementJobType.Name + "\n" + validation.GetReport());
				if (validation.Errors.Count > 0)
				{
					return;
				}
			}
			else if (validation.Warnings.Count > 0)
			{
				GenericJobReplacePatch.Warn("注册警告 | " + originalJobType.Name + " -> " + replacementJobType.Name + "\n" + validation.GetReport());
			}
			else if (validation.Infos.Count > 0)
			{
				GenericJobReplacePatch.Info("注册信息 | " + originalJobType.Name + " -> " + replacementJobType.Name + "\n" + validation.GetReport());
			}
			lock (GenericJobReplacePatch._contextLock)
			{
				if (!GenericJobReplacePatch.activePatchContexts.TryGetValue(method, out var context))
				{
					context = new MethodPatchContext();
					GenericJobReplacePatch.activePatchContexts[method] = context;
				}
				context.JobReplacements[originalJobType] = replacementJobType;
				foreach (KeyValuePair<Type, Type> kvp in validation.ImplicitTypeMapping)
				{
					if (!context.JobReplacements.ContainsKey(kvp.Key))
					{
						context.JobReplacements[kvp.Key] = kvp.Value;
					}
				}
				foreach (FieldInfo oldField in AccessTools.GetDeclaredFields(originalJobType))
				{
					FieldInfo newField = AccessTools.Field(replacementJobType, oldField.Name);
					if (newField != null)
					{
						context.FieldReplacements[oldField] = newField;
						continue;
					}
					GenericJobReplacePatch.Warn("字段缺失警告 | 在新Job " + replacementJobType.Name + " 中未找到字段: " + oldField.Name + "，可能导致 Patch 失败.");
				}
			}
		}

		public static void ClearCache()
		{
			lock (GenericJobReplacePatch._contextLock)
			{
				GenericJobReplacePatch.activePatchContexts.Clear();
				GenericJobReplacePatch.Info("♻\ufe0f 已清理所有 Patch 上下文缓存");
			}
		}

		public static IEnumerable<CodeInstruction> Transpiler(IEnumerable<CodeInstruction> instructions, ILGenerator il, MethodBase originalMethod)
		{
			if (Mod.IsUnloading)
			{
				foreach (CodeInstruction instruction2 in instructions)
				{
					yield return instruction2;
				}
				yield break;
			}
			MethodPatchContext context;
			lock (GenericJobReplacePatch._contextLock)
			{
				if (!GenericJobReplacePatch.activePatchContexts.TryGetValue(originalMethod, out context))
				{
					KeyValuePair<MethodBase, MethodPatchContext> fallbackEntry = GenericJobReplacePatch.activePatchContexts.FirstOrDefault((KeyValuePair<MethodBase, MethodPatchContext> kvp) => kvp.Key.Name == originalMethod.Name && kvp.Key.DeclaringType == originalMethod.DeclaringType);
					if (fallbackEntry.Key != null)
					{
						context = fallbackEntry.Value;
					}
				}
			}
			if (context == null || context.JobReplacements.Count == 0)
			{
				foreach (CodeInstruction instruction3 in instructions)
				{
					yield return instruction3;
				}
				yield break;
			}
			List<CodeInstruction> list = instructions.ToList();
			Dictionary<int, LocalBuilder> localRedirects = new Dictionary<int, LocalBuilder>();
			int replacementCount = 0;
			MethodBody methodBody = originalMethod.GetMethodBody();
			if (methodBody != null)
			{
				foreach (LocalVariableInfo localVar in methodBody.LocalVariables)
				{
					if (localVar.LocalType != null && context.JobReplacements.TryGetValue(localVar.LocalType, out var newType))
					{
						localRedirects[localVar.LocalIndex] = il.DeclareLocal(newType);
					}
				}
			}
			else
			{
				GenericJobReplacePatch.Warn("    ⚠\ufe0f 无法获取方法体局部变量信息: " + originalMethod.Name);
			}
			unchecked
			{
				for (int i = 0; i < list.Count; i++)
				{
					CodeInstruction instruction = list[i];
					if (instruction.opcode == OpCodes.Initobj && instruction.operand is Type initType && context.JobReplacements.TryGetValue(initType, out var newInitType))
					{
						CodeInstruction newInst = new CodeInstruction(OpCodes.Initobj, newInitType);
						newInst.labels = instruction.labels;
						yield return newInst;
						replacementCount++;
						continue;
					}
					int currentLocIndex = GenericJobReplacePatch.GetLocalIndex(instruction);
					if (currentLocIndex >= 0 && localRedirects.TryGetValue(currentLocIndex, out var newLocalBuilder))
					{
						CodeInstruction newInst2 = new CodeInstruction(instruction.opcode, newLocalBuilder);
						newInst2.opcode = GenericJobReplacePatch.ConvertToLongOpCode(instruction.opcode);
						newInst2.labels = instruction.labels;
						newInst2.blocks = instruction.blocks;
						yield return newInst2;
						continue;
					}
					if (instruction.opcode == OpCodes.Stfld && instruction.operand is FieldInfo oldFieldInfo)
					{
						if (context.FieldReplacements.TryGetValue(oldFieldInfo, out var newFieldInfo))
						{
							if (oldFieldInfo.FieldType != newFieldInfo.FieldType)
							{
								MethodInfo bitcastMethod = AccessTools.Method(typeof(GenericJobReplacePatch), "Bitcast").MakeGenericMethod(oldFieldInfo.FieldType, newFieldInfo.FieldType);
								CodeInstruction callInst = new CodeInstruction(OpCodes.Call, bitcastMethod);
								callInst.labels = instruction.labels;
								yield return callInst;
								yield return new CodeInstruction(OpCodes.Stfld, newFieldInfo);
							}
							else
							{
								CodeInstruction newInst3 = new CodeInstruction(instruction.opcode, newFieldInfo);
								newInst3.labels = instruction.labels;
								yield return newInst3;
							}
							continue;
						}
						newFieldInfo = null;
					}
					if ((instruction.opcode == OpCodes.Ldfld || instruction.opcode == OpCodes.Ldflda) && instruction.operand is FieldInfo fieldOp && context.FieldReplacements.TryGetValue(fieldOp, out var newField))
					{
						if (fieldOp.FieldType != newField.FieldType)
						{
							if (!(instruction.opcode == OpCodes.Ldflda))
							{
								MethodInfo bitcastMethod2 = AccessTools.Method(typeof(GenericJobReplacePatch), "Bitcast").MakeGenericMethod(newField.FieldType, fieldOp.FieldType);
								yield return new CodeInstruction(instruction.opcode, newField)
								{
									labels = instruction.labels
								};
								yield return new CodeInstruction(OpCodes.Call, bitcastMethod2);
								continue;
							}
							GenericJobReplacePatch.Warn("    ⚠\ufe0f [Ldflda] 遇到类型不一致的地址加载，这可能导致 IL 指针解引用时的类型安全问题: " + fieldOp.Name);
						}
						yield return new CodeInstruction(instruction.opcode, newField)
						{
							labels = instruction.labels
						};
						continue;
					}
					CodeInstruction finalInst = instruction;
					if (instruction.operand is Type opType && context.JobReplacements.TryGetValue(opType, out var repType))
					{
						finalInst = new CodeInstruction(instruction.opcode, repType);
					}
					else if ((instruction.opcode == OpCodes.Call || instruction.opcode == OpCodes.Callvirt) && instruction.operand is MethodInfo { IsGenericMethod: not false } method)
					{
						Type[] genericArguments = method.GetGenericArguments();
						Type[] newArgs = genericArguments.Select((Type t) => (!context.JobReplacements.ContainsKey(t)) ? t : context.JobReplacements[t]).ToArray();
						if (!genericArguments.SequenceEqual(newArgs))
						{
							try
							{
								MethodInfo newM = method.GetGenericMethodDefinition().MakeGenericMethod(newArgs);
								finalInst = new CodeInstruction(instruction.opcode, newM);
							}
							catch (Exception ex)
							{
								GenericJobReplacePatch.Error("    ❌ 构造泛型方法失败: " + method.Name + ". 错误: " + ex.Message);
							}
						}
					}
					if (finalInst != instruction)
					{
						finalInst.labels = instruction.labels;
						finalInst.blocks = instruction.blocks;
						yield return finalInst;
					}
					else
					{
						yield return instruction;
					}
				}
			}
		}

		private static bool IsLocalVariableOpCode(OpCode op)
		{
			if (!(op == OpCodes.Ldloc) && !(op == OpCodes.Ldloc_S) && !(op == OpCodes.Ldloc_0) && !(op == OpCodes.Ldloc_1) && !(op == OpCodes.Ldloc_2) && !(op == OpCodes.Ldloc_3) && !(op == OpCodes.Ldloca) && !(op == OpCodes.Ldloca_S) && !(op == OpCodes.Stloc) && !(op == OpCodes.Stloc_S) && !(op == OpCodes.Stloc_0) && !(op == OpCodes.Stloc_1) && !(op == OpCodes.Stloc_2))
			{
				return op == OpCodes.Stloc_3;
			}
			return true;
		}

		private static int GetLocalIndex(CodeInstruction i)
		{
			if (!GenericJobReplacePatch.IsLocalVariableOpCode(i.opcode))
			{
				return -1;
			}
			if (i.operand is LocalBuilder lb)
			{
				return lb.LocalIndex;
			}
			if (i.operand is LocalVariableInfo lvi)
			{
				return lvi.LocalIndex;
			}
			if (i.opcode == OpCodes.Ldloc_0 || i.opcode == OpCodes.Stloc_0)
			{
				return 0;
			}
			if (i.opcode == OpCodes.Ldloc_1 || i.opcode == OpCodes.Stloc_1)
			{
				return 1;
			}
			if (i.opcode == OpCodes.Ldloc_2 || i.opcode == OpCodes.Stloc_2)
			{
				return 2;
			}
			if (i.opcode == OpCodes.Ldloc_3 || i.opcode == OpCodes.Stloc_3)
			{
				return 3;
			}
			if (i.operand != null)
			{
				try
				{
					return Convert.ToInt32(i.operand);
				}
				catch
				{
				}
			}
			return -1;
		}

		private static OpCode ConvertToLongOpCode(OpCode op)
		{
			if (op == OpCodes.Ldloc_0 || op == OpCodes.Ldloc_1 || op == OpCodes.Ldloc_2 || op == OpCodes.Ldloc_3 || op == OpCodes.Ldloc_S)
			{
				return OpCodes.Ldloc;
			}
			if (op == OpCodes.Stloc_0 || op == OpCodes.Stloc_1 || op == OpCodes.Stloc_2 || op == OpCodes.Stloc_3 || op == OpCodes.Stloc_S)
			{
				return OpCodes.Stloc;
			}
			if (op == OpCodes.Ldloca_S)
			{
				return OpCodes.Ldloca;
			}
			return op;
		}

		public unsafe static TTo Bitcast<TFrom, TTo>(TFrom source) where TFrom : unmanaged where TTo : unmanaged
		{
			TTo dest = default(TTo);
			UnsafeUtility.MemCpy(UnsafeUtility.AddressOf<TTo>(ref dest), UnsafeUtility.AddressOf<TFrom>(ref source), unchecked((long)UnsafeUtility.SizeOf<TTo>()));
			return dest;
		}

		static GenericJobReplacePatch()
		{
			GenericJobReplacePatch.ModName = "MapExtPDX";
			GenericJobReplacePatch.patchTypeName = "GenericJobReplacePatch";
			GenericJobReplacePatch.activePatchContexts = new Dictionary<MethodBase, MethodPatchContext>();
			GenericJobReplacePatch._contextLock = new object();
		}
	}
}
